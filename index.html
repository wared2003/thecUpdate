Toggle de titre

on stock dans la variable titles les titres récupérés avec un queryselector.
On créer une variable titles_position dans laquelle on stockera la position du toggle

let titles = document.querySelector('#titles');
let titles_position = 0;

on ajoute un écouteur d'evement sur les titres pour détecter le click. En cas de clic on verifie l'etat du toglle et on change le texte en fonction.

    
        titles.addEventListener('click', () => {
            let old_html = titles.innerHTML;
            if (titles_position == 0 ){
                titles.innerHTML = "
Ca c'est un sous-titre
" +
                    "
Ca c'est un titre
";
                titles_position = 1;
            }else{
                titles.innerHTML = "
Ca c'est un titre
" +
                    "
Ca c'est un sous-titre
";
                titles_position = 0;
            }

        })

Selecteur de liste

on récupère dans la variable li tous les items de la liste via un querySelectorAll qui renvoie un array
on récupère dans la variable listbtn le bouton qui controlera la liste

let li = document.querySelectorAll('.list-group-item');
let listbtn = document.querySelector('#list-btn');

on créer et initialise à 0 la variable active_li qui stockera l'élement de la liste actif
on initialise également la variable back_to_botom qui stockera la direction du selecteur de la liste

let active_li = 0;
let back_to_botom = true;

on ajoute un écoueteur d'évenement sur le le bouton stocker dans listbtn en fonction de la direction et la position du sélecteur on itère dans la liste on retire la class active a l'élement active et on ajoute à lélement suivant la class active 

listbtn.addEventListener('click', () => {

    if (active_li < 4 && back_to_botom == true) {
        li[active_li].className = 'list-group-item';
        li[active_li + 1].className = 'list-group-item active';
        active_li++;
    } else {
        li[active_li].className = 'list-group-item';
        li[active_li - 1].className = 'list-group-item active';
        active_li--;
        if (active_li != 0) {
            back_to_botom = false;
        } else {
            back_to_botom = true;
        }
    }
})
